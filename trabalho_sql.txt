--3. Mostre o nome da cidade que tem IPVA em atraso (mesmo que tenha mais de um veículo em atraso em uma cidade, o nome da cidade deve ser mostrado apenas 1 vez)
SELECT nome, count(nome) quantidade_de_carros
FROM cidade
INNER JOIN veiculos
ON veiculos.codigo_cidade=cidade.codigo
WHERE vencimento_ipva < TO_DATE('12/11/2019','DD/MM/YYYY')
GROUP BY cidade.nome

--4. Mostre todos os veículos de um determinado tipo (você escolhe o tipo que fará a restrição)
SELECT *
FROM veiculos
WHERE tipo='CAMINHÃO'

--5.Informe o nome do proprietário e o número da placa para todos os registrados cadastrados

SELECT proprietarios.nome, veiculos.placa AS placa 
FROM veiculos 
INNER JOIN proprietarios
ON proprietarios.cnh=veiculos.cnh_proprietarios
ORDER BY proprietarios.nome

--6. Mostre o número da placa e valor de multas para cada placa

SELECT veiculos.placa, SUM(multas.valor) AS soma_das_multas
FROM veiculos
INNER JOIN multas
ON multas.chassis=veiculos.chassis
GROUP BY veiculos.placa
ORDER BY soma_das_multas ASC

7. Mostre a quantidade de veículos por cor

SELECT cor.nome AS cor_do_veiculo, COUNT(veiculos.codigo_cor) AS quantidade 
FROM veiculos
INNER JOIN cor
ON cor.codigo=veiculos.codigo_cor
GROUP BY cor.nome
ORDER BY cor.nome ASC

8. Mostre a quantidade de veículos por proprietário, ordenando da menor para a maior quantidade

SELECT proprietarios.nome, COUNT(veiculos.cnh_proprietarios) AS quantidade_de_veiculos
FROM proprietarios
INNER JOIN veiculos
ON veiculos.cnh_proprietarios=proprietarios.cnh
GROUP BY proprietarios.nome
ORDER BY quantidade_de_veiculos ASC

9.	Crie uma nova tabela com o nome VeículosBaixados. Essa tabela guardará as informações de veículos baixados,
 aqueles veículos que não poderão mais circular. Essa tabela deve ter a informação do veículo 
que será chave estrangeira da tabela veículos, a data que o veículo foi baixado e um campo observação 
que irá conter o motivo da baixa

CREATE TABLE VeiculosBaixados
( codigo NUMBER(20) CONSTRAINT  pk_codigo PRIMARY KEY, 
  data_baixado DATE,
  observacao VARCHAR2(50),
  veiculo_chassis VARCHAR2(15),
  CONSTRAINT fk_veiculos_chassis FOREIGN KEY (veiculo_chassis) REFERENCES veiculos(chassis)
);

INNER JOIN veiculosbaixados
ON veiculosbaixados.veiculo_chassis=veiculos.chassis
GROUP BY cidade.nome
ORDER BY Soma_dos_baixados DESC
